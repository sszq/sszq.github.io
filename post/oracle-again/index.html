Oracle Again

November 15, 2020
SQL

tablespace

-- 创建表空间
create tablespace user1 datafile 'C:\app\oracle\oradata\maximo\user1_data.dbf' size 20M;
-- 创建表空间，可扩展性
create tablespace user2 datafile 'C:\app\oracle\oradata\maximo\user2_data.dbf' size 100M autoextend on;
-- 创建表空间，关闭自动扩展
create tablespace user22 datafile 'C:\app\oracle\oradata\maximo\user22_date.dbf' size 500M autoextend off;
-- 指定数据文件的增长幅度
create tablespace user3 datafile 'C:\app\oracle\oradata\maximo\user3_data_dbf' size 20M autoextend on next 5M;
-- 指定数据文件的最大容量
create tablespace user4 datafile 'C:\app\oracle\oradata\maximo\user4_data_dbf' size 20M autoextend on next 5M maxsize 500M;
-- 指定数据文件的容量为不限制
create tablespace user42 datafile 'C:\app\oracle\oradata\maximo\user42_data_dbf' size 20M autoextend on next 5M maxsize unlimited;
-- 为一个表空间创建多个数据文件
create tablespace multiple_data_file datafile 'C:\app\oracle\oradata\maximo\data_1.dbf' size 1M, 'C:\app\oracle\oradata\maximo\data_2.dbf' size 5M;
-- 查看表空间文件
select * from dba_data_files;
-- 查看所有的表空间
select * from dba_tablespaces;
-- 查看用户默认的表空间
select user_id,username,default_tablespace from dba_users;
-- 修改表空间名称
alter database default tablespace user1;
-- 修改表空间名称（只更改数据库内的名称，资源管理器中数据文件的文件名称并不会改变）
alter tablespace user2 rename to user20;
-- 删除表空间（仅删除其在数据库中的记录）
drop tablespace user20;
-- 删除表空间（并删除资源管理器中的数据文件）[很奇怪MULTIPLE_DATA_FILE的数据文件没有删除掉]
drop tablespace user3 including contents and datafiles;
drop tablespace MULTIPLE_DATA_FILE including contents and datafiles;

table


select d.* from dba_tables d where d.TABLE_NAME like 'UD%';

-- 创建表
create table t_user(
user_id number not null,
user_name varchar(20) not null,
user_email varchar(20)
)

select * from  t_users for update;
-- 修改表列，列更名时，不会影响数据库中的记录
alter table t_user rename column emial to user_email;
-- 更改列的属性
alter table t_user modify (user_name varchar2(10)) 
-- 同时更改多个列属性
alter table t_user modify(user_name varchar2(20), user_email varchar(20))
-- 添加列
alter table t_user add (address varchar(20))
-- 删除列（有数据也会删除），因为可能存在某些约束，需要添加column
alter table t_user drop column address;
-- 更改表名
alter table t_user rename to t_users;

create table t_class(a int, b int, c varchar(10) );
-- 删除数据库表
drop table t_class;
-- 删除数据库表数据
delete from t_class;
-- 删除数据库表（存在外键引用时）
drop table t_class cascade constraint;
