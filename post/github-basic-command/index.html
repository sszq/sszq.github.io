# defaut click three enter
ssh-keygen -t -rsa -C "1004170070@qq.com"

cat /mnt/c/User/Administrator/.ssh/id_rsa.pub

ssh -T git@github.com

config you user info

git config --global user.name "sszq"
git config --global user.email "1004170070@qq.com"

push local project to github repository

# to the directory what you project path
cd ianun-crowlers
# initialization the localhost project that you want to upload to github
git init
# add all file to localhost
git add .
# commit all files to local repository
git commit -m "initialization project"
# 
# git branch -M main
# connection local project to github project
git remote add origin git@github.com:sszq/ianun-scrowlers.git
# push to github repository
git push -u origin main

git remote add origin git@github.com:sszq/ianun-scrowlers.git
git branch -M main
